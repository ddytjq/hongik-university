<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="Arial" size="2">Create a floating toolbar.</font></td>  </tr>
</table><br>
<table border="0" width="100%" bgcolor="#FFCC00">
  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip Code</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="arial" size="-1" ></font><pre><font color="#00007F">Option</font> <font color="#00007F">Explicit</font>

<font color="#007F00">'These variables save the toolbar location while unloaded</font>
<font color="#00007F">Public</font> g_xToolbarPos <font color="#00007F">As</font> <font color="#00007F">Integer</font>
<font color="#00007F">Public</font> g_yToolbarPos <font color="#00007F">As</font> <font color="#00007F">Integer</font>

<font color="#007F00">'Toolbar visible state</font>
<font color="#00007F">Public</font> g_bToolbarVisible <font color="#00007F">As</font> Boolean</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica">

</font><pre><font color="#00007F">Option</font> <font color="#00007F">Explicit</font>

<font color="#007F00">'Set values for Image1 bitmap</font>
<font color="#00007F">Private</font> <font color="#00007F">Const</font> IMAGE_COLS = 15
<font color="#00007F">Private</font> <font color="#00007F">Const</font> IMAGE_ROWS = 2
<font color="#00007F">Private</font> <font color="#00007F">Const</font> STATE_UNSELECTED = 0
<font color="#00007F">Private</font> <font color="#00007F">Const</font> STATE_SELECTED = 1

<font color="#00007F">Private</font> m_ImageWidth <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Private</font> m_ImageHeight <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Private</font> m_xSpacer <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Private</font> m_ySpacer <font color="#00007F">As</font> <font color="#00007F">Long</font>

<font color="#007F00">'Indicates which tool is currently selected</font>
<font color="#00007F">Private</font> m_nCurrTool <font color="#00007F">As</font> <font color="#00007F">Integer</font>

<font color="#007F00">'Tracks parent form</font>
<font color="#00007F">Private</font> m_frmParent <font color="#00007F">As</font> Form

<font color="#007F00">'Windows API declarations</font>
<font color="#00007F">Private</font> <font color="#00007F">Declare</font> <font color="#00007F">Function</font> SetWindowWord <font color="#00007F">Lib</font> "user32" (<font color="#00007F">ByVal</font> hwnd _
<font color="#00007F">As</font> <font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> nIndex <font color="#00007F">As</font> <font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> wNewWord <font color="#00007F">As</font> <font color="#00007F">Long</font>) <font color="#00007F">As</font> <font color="#00007F">Long</font>

<font color="#00007F">Private</font> <font color="#00007F">Const</font> SWW_HPARENT = -8

<font color="#00007F">Public</font> <font color="#00007F">Sub</font> LoadToolbar(frm <font color="#00007F">As</font> Form)

<font color="#007F00">'Save reference to parent form</font>
<font color="#00007F">Set</font> m_frmParent = frm

<font color="#007F00">'Display toolbar</font>
Show

<font color="#007F00">'Make toolbar a child window of the main window so that</font>
<font color="#007F00">'it always remains on top of the main window, but not</font>
<font color="#007F00">'other windows</font>
<font color="#00007F">Call</font> SetWindowWord(hwnd, SWW_HPARENT, m_frmParent.hwnd)

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Public</font> <font color="#00007F">Sub</font> UnloadToolbar()
<font color="#007F00">'Note: Unload event performs clean up</font>
Unload <font color="#00007F">Me</font>
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Public</font> <font color="#00007F">Property</font> <font color="#00007F">Get</font> Tool() <font color="#00007F">As</font> <font color="#00007F">Integer</font>
Tool = m_nCurrTool
<font color="#00007F">End</font> <font color="#00007F">Property</font>

<font color="#007F00">'Public property to set selected tool</font>
<font color="#00007F">Public</font> <font color="#00007F">Property</font> <font color="#00007F">Let</font> Tool(nTool <font color="#00007F">As</font> <font color="#00007F">Integer</font>)
SetCurrentTool nTool
<font color="#00007F">End</font> <font color="#00007F">Property</font>


<font color="#00007F">Private</font> <font color="#00007F">Sub</font> SetCurrentTool(nNewTool <font color="#00007F">As</font> <font color="#00007F">Integer</font>)

<font color="#00007F">If</font> nNewTool <> m_nCurrTool <font color="#00007F">Then</font>

  <font color="#007F00">'Unselect old tool</font>
  PaintTool m_nCurrTool, STATE_UNSELECTED

  <font color="#007F00">'Set new tool number</font>
  m_nCurrTool = nNewTool

  <font color="#007F00">'Select new tool</font>
  PaintTool m_nCurrTool, STATE_SELECTED

<font color="#00007F">End</font> <font color="#00007F">If</font>

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> PaintTool(nTool <font color="#00007F">As</font> <font color="#00007F">Integer</font>, nState <font color="#00007F">As</font> <font color="#00007F">Integer</font>)

<font color="#007F00">'Paint specified cell to form</font>
PaintPicture Image1, _
m_xSpacer + ((nTool <font color="#00007F">Mod</font> 2) * m_ImageWidth), _
m_ySpacer + ((nTool \ 2) * m_ImageHeight), _
m_ImageWidth, m_ImageHeight, _
nTool * m_ImageWidth, nState * m_ImageHeight, _
m_ImageWidth, m_ImageHeight
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Form_Load()

m nWidth <font color="#00007F">As</font> <font color="#00007F">Long</font>, nHeight <font color="#00007F">As</font> <font color="#00007F">Long</font>

<font color="#007F00">'Cache image cell size for speed</font>
m_ImageWidth = Image1.Width \ IMAGE_COLS
m_ImageHeight = Image1.Height \ IMAGE_ROWS

<font color="#007F00">'Create spacer around toolbar buttons</font>
m_xSpacer = (3 * Screen.TwipsPerPixelX)
m_ySpacer = (3 * Screen.TwipsPerPixelY)

<font color="#007F00">'Size form to fit buttons</font>
nWidth = (m_ImageWidth * 2) + (m_xSpacer * 2)
nHeight = (m_ImageWidth * ((IMAGE_COLS + 1) \ 2)) + (m_ySpacer * 2)

Width = nWidth + (Width - ScaleWidth)
Height = nHeight + (Height - ScaleHeight)

<font color="#007F00">'Position toolbar at remembered location</font>
<font color="#00007F">Left</font> = g_xToolbarPos: Top = g_yToolbarPos

<font color="#007F00">'Indicate toolbar is visible</font>
g_bToolbarVisible = True

<font color="#007F00">'Indicate which tool is selected</font>
<font color="#00007F">Call</font> SetCurrentTool(0)

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Form_Paint()
<font color="#00007F">Dim</font> i <font color="#00007F">As</font> <font color="#00007F">Integer</font>, nState <font color="#00007F">As</font> <font color="#00007F">Integer</font>

<font color="#00007F">For</font> i = 0 <font color="#00007F">To</font> (IMAGE_COLS - 1)

<font color="#00007F">If</font> i = m_nCurrTool <font color="#00007F">Then</font>
  nState = STATE_SELECTED
<font color="#00007F">Else</font>
  nState = STATE_UNSELECTED
<font color="#00007F">End</font> <font color="#00007F">If</font>

PaintTool i, nState

<font color="#00007F">Next</font> i

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Form_MouseDown(button <font color="#00007F">As</font> <font color="#00007F">Integer</font>, Shift <font color="#00007F">As</font> _
<font color="#00007F">Integer</font>, X <font color="#00007F">As</font> <font color="#00007F">Single</font>, Y <font color="#00007F">As</font> <font color="#00007F">Single</font>)
<font color="#00007F">Dim</font> nTool <font color="#00007F">As</font> <font color="#00007F">Integer</font>

<font color="#00007F">If</font> button = vbLeftButton <font color="#00007F">Then</font>
  
  <font color="#007F00">'Adjust for border around buttons</font>
  X = X - m_xSpacer
  Y = Y - m_ySpacer

  <font color="#007F00">'Prevent false hits</font>
  <font color="#00007F">If</font> X >= 0 <font color="#00007F">And</font> X < (m_ImageWidth * 2) <font color="#00007F">And</font> _
  Y >= 0 <font color="#00007F">And</font> Y < (m_ImageHeight * ((IMAGE_COLS + 1) \ 2)) <font color="#00007F">Then</font>
    nTool = ((Y \ m_ImageHeight) * 2) + (X \ m_ImageWidth)
    
    <font color="#007F00">'Set new toolnumber if valid</font>
    <font color="#00007F">If</font> nTool < IMAGE_COLS <font color="#00007F">Then</font>
      SetCurrentTool nTool
    <font color="#00007F">End</font> <font color="#00007F">If</font>
  <font color="#00007F">End</font> <font color="#00007F">If</font>
<font color="#00007F">End</font> <font color="#00007F">If</font>

<font color="#00007F">End</font> Sub</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica"></font></td></tr></table>
