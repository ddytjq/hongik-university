// TB 06/19/00
function validate(){
var error_str='';
var error_focus_set=0;

// confirm user name is present
var str=document.support_form.name;
if (str.value.length<1){
error_str+='\nPlease include your full name.';
if (!error_focus_set) error_focus_set=str;
}

// validate user email address present and structured
var str=document.support_form.email;
if (str.value.length<1){
error_str+='\nPlease include your e-mail address.';
if (!error_focus_set) error_focus_set=str;
}
else{
if (str.value.indexOf("@")<1){
error_str+='\nPlease enter a valid e-mail address.';
if (!error_focus_set) error_focus_set=str;
}
}

// validate site control
var str=document.support_form.site;
if (str.value.length<1){
error_str+='\nPlease include the URL to your web site.';
if (!error_focus_set) error_focus_set=str;
}

// validate question is present
var str=document.support_form.suggestion;
if (str.value.length<1){
error_str+='\nPlease include your suggestion.';
if (!error_focus_set) error_focus_set=str[0];
}

// submit or display errors
if (error_str.length>0){
alert(error_str);
if (error_focus_set) error_focus_set.focus();
return false;
}
else{
adjust_oneliner();
document.support_form.submit();
}
}

function adjust_oneliner(){
// purpose, to tack the first 50 characters of the suggestion to the subject header for the email.

var str='';
var mobj=document.support_form.suggestion.value;
for (var index=0; (index<mobj.length&&index<50); index++){
str+=mobj.charAt(index);
}
document.support_form.subject.value+=': '+str;
//alert(document.support_form.subject.value);
}
