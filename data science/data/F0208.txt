<font color="#00007F">Public</font> <font color="#00007F">Function</font> PostInfo$(srv$, script$, postdat$)

  <font color="#00007F">Dim</font> hInternetOpen <font color="#00007F">As</font> <font color="#00007F">Long</font>
  <font color="#00007F">Dim</font> hInternetConnect <font color="#00007F">As</font> <font color="#00007F">Long</font>
  <font color="#00007F">Dim</font> hHttpOpenRequest <font color="#00007F">As</font> <font color="#00007F">Long</font>
  <font color="#00007F">Dim</font> bRet <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
  
  hInternetOpen = 0
  hInternetConnect = 0
  hHttpOpenRequest = 0
  
  <font color="#007F00">'Use registry access settings.</font>
  <font color="#00007F">Const</font> INTERNET_OPEN_TYPE_PRECONFIG = 0
  hInternetOpen = InternetOpen("http generic", _
                  INTERNET_OPEN_TYPE_PRECONFIG, _
                  vbNullString, _
                  vbNullString, _
                  0)
  
  <font color="#00007F">If</font> hInternetOpen <> 0 <font color="#00007F">Then</font>
     <font color="#007F00">'Type of service to access.</font>
     <font color="#00007F">Const</font> INTERNET_SERVICE_HTTP = 3
     <font color="#00007F">Const</font> INTERNET_DEFAULT_HTTP_PORT = 80
     <font color="#007F00">'Change the server to your server name</font>
     hInternetConnect = InternetConnect(hInternetOpen, _
                        srv$, _
                        INTERNET_DEFAULT_HTTP_PORT, _
                        vbNullString, _
                        "HTTP/1.0", _
                        INTERNET_SERVICE_HTTP, _
                        0, _
                        0)
  
     <font color="#00007F">If</font> hInternetConnect <> 0 <font color="#00007F">Then</font>
      <font color="#007F00">'Brings the data across the wire even if it locally cached.</font>
       <font color="#00007F">Const</font> INTERNET_FLAG_RELOAD = &H80000000
       hHttpOpenRequest = HttpOpenRequest(hInternetConnect, _
                           "POST", _
                           script$, _
                           "HTTP/1.0", _
                           vbNullString, _
                           0, _
                           INTERNET_FLAG_RELOAD, _
                           0)
  
        <font color="#00007F">If</font> hHttpOpenRequest <> 0 <font color="#00007F">Then</font>
           <font color="#00007F">Dim</font> sHeader <font color="#00007F">As</font> <font color="#00007F">String</font>
           <font color="#00007F">Const</font> HTTP_ADDREQ_FLAG_ADD = &H20000000
           <font color="#00007F">Const</font> HTTP_ADDREQ_FLAG_REPLACE = &H80000000
  sHeader = "Content-Type: application/x-www-form-urlencoded" _
             & vbCrLf
           bRet = HttpAddRequestHeaders(hHttpOpenRequest, _
             sHeader, <font color="#00007F">Len</font>(sHeader), HTTP_ADDREQ_FLAG_REPLACE _
             <font color="#00007F">Or</font> HTTP_ADDREQ_FLAG_ADD)
  
           <font color="#00007F">Dim</font> lpszPostData <font color="#00007F">As</font> <font color="#00007F">String</font>
           <font color="#00007F">Dim</font> lPostDataLen <font color="#00007F">As</font> <font color="#00007F">Long</font>
  
           lpszPostData = postdat$
           lPostDataLen = <font color="#00007F">Len</font>(lpszPostData)
           bRet = HttpSendRequest(hHttpOpenRequest, _
                  vbNullString, _
                  0, _
                  lpszPostData, _
                  lPostDataLen)
  
           <font color="#00007F">Dim</font> bDoLoop             <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
           <font color="#00007F">Dim</font> sReadBuffer         <font color="#00007F">As</font> <font color="#00007F">String</font> * 2048
           <font color="#00007F">Dim</font> lNumberOfBytesRead  <font color="#00007F">As</font> <font color="#00007F">Long</font>
           <font color="#00007F">Dim</font> sBuffer             <font color="#00007F">As</font> <font color="#00007F">String</font>
           bDoLoop = True
           <font color="#00007F">While</font> bDoLoop
            sReadBuffer = vbNullString
            bDoLoop = InternetReadFile(hHttpOpenRequest, _
               sReadBuffer, <font color="#00007F">Len</font>(sReadBuffer), lNumberOfBytesRead)
            sBuffer = sBuffer & _
                 <font color="#00007F">Left</font>(sReadBuffer, lNumberOfBytesRead)
            <font color="#00007F">If</font> <font color="#00007F">Not</font> <font color="#00007F">CBool</font>(lNumberOfBytesRead) <font color="#00007F">Then</font> bDoLoop = False
           <font color="#00007F">Wend</font>
           PostInfo = sBuffer
           bRet = InternetCloseHandle(hHttpOpenRequest)
        <font color="#00007F">End</font> <font color="#00007F">If</font>
        bRet = InternetCloseHandle(hInternetConnect)
     <font color="#00007F">End</font> <font color="#00007F">If</font>
     bRet = InternetCloseHandle(hInternetOpen)
  <font color="#00007F">End</font> I</p>
<p>nd <font color="#00007F">Function</font>

<font color="#00007F">Public</font> <font color="#00007F">Sub</font> SplitAddr(<font color="#00007F">ByVal</font> addr$, srv$, script$)
<font color="#007F00">'Inputs: The full url including http://</font>
<font color="#007F00">' Two variables that will be changed</font>
<font color="#007F00">'</font>
<font color="#007F00">'Returns: Splits the addr$ var into the server name</font>
<font color="#007F00">' and the script path</font>

  <font color="#00007F">Dim</font> i%

  i = <font color="#00007F">InStr</font>(addr$, "/")
  srv$ = <font color="#00007F">Mid</font>(addr$, i + 2, <font color="#00007F">Len</font>(addr$) - (i + 1))
  i = <font color="#00007F">InStr</font>(srv$, "/")
  script$ = <font color="#00007F">Mid</font>(srv$, i, <font color="#00007F">Len</font>(srv$) + 1 - i)
  srv$ = <font color="#00007F">Left</font>$(srv$, i - 1)

<font color="#00007F">End</font> <font color="#00007F">Sub</font>
