function validate_email(email) {
// Validate email address input
// returns true, if email valid
  var valid_email = 0;
  for(i=0; i<=email.length; i++){
    if (email.substring(i, i+1)==" ") { valid_email=99; }
    if (email.substring(i, i+1)=="@" && valid_email==0) { valid_email=1; }
    if (email.substring(i, i+1)=="." && valid_email==1) { valid_email=2; }
  }
  if (valid_email==2) {
    return true;
  }else {
    return false;
  }    
}
