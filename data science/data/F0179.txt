<font color="#00007F">Public</font> <font color="#00007F">Sub</font> TextEffect(obj <font color="#00007F">As</font> <font color="#00007F">Object</font>, <font color="#00007F">ByVal</font> sText <font color="#00007F">As</font> <font color="#00007F">String</font>, _
<font color="#00007F">ByVal</font> lX <font color="#00007F">As</font> <font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> lY <font color="#00007F">As</font> <font color="#00007F">Long</font>, Optional <font color="#00007F">ByVal</font> bLoop _
<font color="#00007F">As</font> <font color="#00007F">Boolean</font> = False, Optional <font color="#00007F">ByVal</font> lStartSpacing <font color="#00007F">As</font> <font color="#00007F">Long</font> = 128, _
Optional <font color="#00007F">ByVal</font> lEndSpacing <font color="#00007F">As</font> <font color="#00007F">Long</font> = -1, Optional <font color="#00007F">ByVal</font> oColor _

   <font color="#00007F">Dim</font> x                <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> lLen             <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> hBrush           <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Static</font> tR            <font color="#00007F">As</font> RECT
   <font color="#00007F">Dim</font> iDir             <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> bNotFirstTime    <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
   <font color="#00007F">Dim</font> lTime            <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> lIter            <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> bSlowDown        <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
   <font color="#00007F">Dim</font> lCOlor           <font color="#00007F">As</font> <font color="#00007F">Long</font>
   <font color="#00007F">Dim</font> bDoIt            <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
   
   lhDC = obj.hdc
   iDir = -1
   i = lStartSpacing
   tR.<font color="#00007F">Left</font> = lX: tR.Top = lY: tR.<font color="#00007F">Right</font> = lX: tR.Bottom = lY
   OleTranslateColor oColor, 0, lCOlor
   
   hBrush = CreateSolidBrush(GetSysColor(COLOR_BTNFACE))
   lLen = <font color="#00007F">Len</font>(sText)
   
   SetTextColor lhDC, lCOlor
   bDoIt = True
   
   <font color="#00007F">Do</font> <font color="#00007F">While</font> bDoIt
      lTime = timeGetTime
      <font color="#00007F">If</font> (i < -3) <font color="#00007F">And</font> <font color="#00007F">Not</font> (bLoop) <font color="#00007F">And</font> <font color="#00007F">Not</font> (bSlowDown) <font color="#00007F">Then</font>
         bSlowDown = True
         iDir = 1
         lIter = (i + 4)
      <font color="#00007F">End</font> <font color="#00007F">If</font>
      <font color="#00007F">If</font> (i > 128) <font color="#00007F">Then</font> iDir = -1
      <font color="#00007F">If</font> <font color="#00007F">Not</font> (bLoop) <font color="#00007F">And</font> iDir = 1 <font color="#00007F">Then</font>
         <font color="#00007F">If</font> (i = lEndSpacing) <font color="#00007F">Then</font>
            <font color="#007F00">' <font color="#00007F">Stop</font></font>
            bDoIt = False
         <font color="#00007F">Else</font>
            lIter = lIter - 1
            <font color="#00007F">If</font> (lIter <= 0) <font color="#00007F">Then</font>
               i = i + iDir
               lIter = (i + 4)
            <font color="#00007F">End</font> <font color="#00007F">If</font>
         <font color="#00007F">End</font> <font color="#00007F">If</font>
      <font color="#00007F">Else</font>
         i = i + iDir
      <font color="#00007F">End</font> <font color="#00007F">If</font>
      
      FillRect lhDC, tR, hBrush
      x = 32 - (i * lLen)
      SetTextCharacterExtra lhDC, i
      DrawText lhDC, sText, lLen, tR, DT_CALCRECT
      tR.<font color="#00007F">Right</font> = tR.<font color="#00007F">Right</font> + 4
      <font color="#00007F">If</font> (tR.<font color="#00007F">Right</font> > obj.ScaleWidth \ Screen.TwipsPerPixelX) <font color="#00007F">Then</font> _
      tR.<font color="#00007F">Right</font> = obj.ScaleWidth \ Screen.TwipsPerPixelX
      DrawText lhDC, sText, lLen, tR, DT_LEFT
      obj.Refresh
      
      <font color="#00007F">Do</font>
         <font color="#00007F">DoEvents</font>
         <font color="#00007F">If</font> obj.Visible = False <font color="#00007F">Then</font> <font color="#00007F">Exit</font> <font color="#00007F">Sub</font>
      <font color="#00007F">Loop</font> <font color="#00007F">While</font> (timeGetTime - lTime) < 20
   
   <font color="#00007F">Loop</font>
