<p><font face="Arial, Helvetica" size="+1"><b>Type Ahead Combo Box</b></font></p>
<p><i>Tip by <a href="/php-bin/authorbio.php?authorid=53&associd=465&assoctype=3">Brent</a></i></p>

<table border="0" width="100%" bgcolor="#FFCC00">
  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip description</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="Arial" size="2">Would you like to force users to select an itme from your combo box list and also allow them to just type the first few cahracters of a list item to bring the desired item into focus?  Here is the API and code to do it.</font></td>  </tr>
</table><br>
<table border="0" width="100%" bgcolor="#FFCC00">
  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip Code</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="arial" size="-1" ><p></font><pre>
<font color="#00007F">Private</font> <font color="#00007F">Sub</font> MyCombo_KeyPress(KeyAscii <font color="#00007F">As</font> <font color="#00007F">Integer</font>)
    KeyAscii = AutoFind(MyCombo, KeyAscii, False)
<font color="#00007F">End</font> <font color="#00007F">Sub</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica">
<p>Put this in the Declarations Section:</p>

</font><pre>
<font color="#00007F">Public</font> <font color="#00007F">Const</font> CB_FINDSTRING = &H14C
<font color="#00007F">Public</font> <font color="#00007F">Const</font> CB_ERR = (-1)

<font color="#00007F">Declare</font> <font color="#00007F">Function</font> SendMessage <font color="#00007F">Lib</font> "user32" <font color="#00007F">Alias</font> _ 
                                 "SendMessageA" _ 
                                 (<font color="#00007F">ByVal</font> hWnd <font color="#00007F">As</font> <font color="#00007F">Long</font>, _ 
                                  <font color="#00007F">ByVal</font> wMsg <font color="#00007F">As</font> <font color="#00007F">Long</font>, _
                                  <font color="#00007F">ByVal</font> wParam <font color="#00007F">As</font> <font color="#00007F">Long</font>, _ 
                                  lParam <font color="#00007F">As</font> <font color="#00007F">Any</font>) <font color="#00007F">As</font> <font color="#00007F">Long</font>

<font color="#00007F">Public</font> <font color="#00007F">Function</font> AutoFind(ByRef cboCurrent <font color="#00007F">As</font> ComboBox, _ 
                         <font color="#00007F">ByVal</font> KeyAscii <font color="#00007F">As</font> <font color="#00007F">Integer</font>, _ 
                         Optional <font color="#00007F">ByVal</font> LimitToList <font color="#00007F">As</font> <font color="#00007F">Boolean</font> = False)
        
<font color="#00007F">Dim</font> lCB <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> sFindString <font color="#00007F">As</font> <font color="#00007F">String</font>

<font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">GoTo</font> Err_Handler
    <font color="#00007F">If</font> KeyAscii = 8 <font color="#00007F">Then</font>
        <font color="#00007F">If</font> cboCurrent.SelStart <= 1 <font color="#00007F">Then</font>
            cboCurrent = ""
            AutoFind = 0
            <font color="#00007F">Exit</font> <font color="#00007F">Function</font>
        <font color="#00007F">End</font> <font color="#00007F">If</font>
        <font color="#00007F">If</font> cboCurrent.SelLength = 0 <font color="#00007F">Then</font>
            sFindString = <font color="#00007F">UCase</font>(<font color="#00007F">Left</font>(cboCurrent, <font color="#00007F">Len</font>(cboCurrent) - 1))
        <font color="#00007F">Else</font>
            sFindString = <font color="#00007F">Left</font>$(cboCurrent.Text, cboCurrent.SelStart - 1)
        <font color="#00007F">End</font> <font color="#00007F">If</font>
    <font color="#00007F">ElseIf</font> KeyAscii < 32 <font color="#00007F">Or</font> KeyAscii > 127 <font color="#00007F">Then</font>
        <font color="#00007F">Exit</font> <font color="#00007F">Function</font>
    <font color="#00007F">Else</font>
        <font color="#00007F">If</font> cboCurrent.SelLength = 0 <font color="#00007F">Then</font>
            sFindString = <font color="#00007F">UCase</font>(cboCurrent.Text & <font color="#00007F">Chr</font>$(KeyAscii))
        <font color="#00007F">Else</font>
            sFindString = <font color="#00007F">Left</font>$(cboCurrent.Text, cboCurrent.SelStart) & <font color="#00007F">Chr</font>$(KeyAscii)
        <font color="#00007F">End</font> <font color="#00007F">If</font>
    <font color="#00007F">End</font> <font color="#00007F">If</font>
    lCB = SendMessage(cboCurrent.hWnd, CB_FINDSTRING, -1, <font color="#00007F">ByVal</font> sFindString)

    <font color="#00007F">If</font> lCB <> CB_ERR <font color="#00007F">Then</font>
        cboCurrent.ListIndex = lCB
        cboCurrent.SelStart = <font color="#00007F">Len</font>(sFindString)
        cboCurrent.SelLength = <font color="#00007F">Len</font>(cboCurrent.Text) - cboCurrent.SelStart
        AutoFind = 0
    <font color="#00007F">Else</font>
        <font color="#00007F">If</font> LimitToList = True <font color="#00007F">Then</font>
            AutoFind = 0
        <font color="#00007F">Else</font>
            AutoFind = KeyAscii
        <font color="#00007F">End</font> <font color="#00007F">If</font>
    <font color="#00007F">End</font> <font color="#00007F">If</font>
        
<font color="#00007F">End</font> <font color="#00007F">Function</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica"></p></font></td></tr></table>
