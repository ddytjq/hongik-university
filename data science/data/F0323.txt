<p><font face="Arial, Helvetica" size="+1"><b>Make a reusable database class</b></font></p>
<p><i>By <a href="/php-bin/authorbio.php?authorid=31&associd=59&assoctype=2">Sam Huggill</a></i></p>
<p><font face="Arial, Helvetica" size="+1"><b>Introduction</b></font></p>

<p>
This database class might need you to change one db path
setting (which could be improved by using a public variable). Otherwise it contains
prettry much the same functions as the previous article apart from the addition of the
repair and compact functions.
</p>

<p>
Add a new class module and rename it cDataFunc. Copy the
following code into it:
</p>

</font><pre>  
<font color="#00007F">Option</font> <font color="#00007F">Explicit</font>

<font color="#007F00">'Public members of this module</font>
<font color="#00007F">Public</font> iID <font color="#00007F">As</font> <font color="#00007F">Integer</font>
<font color="#00007F">Private</font> db <font color="#00007F">As</font> Database
<font color="#00007F">Private</font> RS <font color="#00007F">As</font> Recordset

<font color="#00007F">Public</font> <font color="#00007F">Sub</font> OpenDB(dbName <font color="#00007F">As</font> <font color="#00007F">String</font>)
   
  <font color="#007F00">'Open the database</font>
  <font color="#00007F">Set</font> db = OpenDatabase(dbName)

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Function</font> GetNextRecord(sRecordSet <font color="#00007F">As</font> <font color="#00007F">String</font>, sField <font color="#00007F">As</font> <font color="#00007F">String</font>, iPos <font color="#00007F">As</font> <font color="#00007F">Integer</font>) <font color="#00007F">As</font> <font color="#00007F">String</font>

  <font color="#007F00">'Open the recordset and select the next record</font>
  <font color="#00007F">Set</font> RS = db.OpenRecordset(&quot;SELECT * FROM &quot; &amp; sRecordSet &amp; &quot; WHERE _  
    &quot; &amp; sField &amp; &quot; =&quot; &amp; iPos, dbOpenDynaset)

  <font color="#007F00">'Return the field value</font>
  iID = &quot;&quot; &amp; RS.Fields(sField).Value

  <font color="#007F00">'Close the recordset</font>
  RS.<font color="#00007F">Close</font>

<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Function</font> UpDate(sRecordSet <font color="#00007F">As</font> <font color="#00007F">String</font>, sField <font color="#00007F">As</font> <font color="#00007F">String</font>, _  
  sValue <font color="#00007F">As</font> <font color="#00007F">String</font>, sFieldToUpdate <font color="#00007F">As</font> <font color="#00007F">String</font>, sValueToUpdate <font color="#00007F">As</font> <font color="#00007F">String</font>)

  <font color="#007F00">'Open the Contact table</font>
  <font color="#00007F">Set</font> RS = db.OpenRecordset(&quot;SELECT * FROM &quot; &amp; sRecordSet _  
    &amp; &quot; WHERE &quot; &amp; sField &amp; &quot; = &quot; &amp; sValue, dbOpenDynaset)

  <font color="#00007F">With</font> RS

      <font color="#007F00">'Set it in edit more</font>
      .Edit

      <font color="#007F00">'Update the following fields</font>
      .Fields(sFieldToUpdate).Value = sValueToUpdate

      <font color="#007F00">'Update the table</font>
      .UpDate

      <font color="#007F00">'Close it</font>
      .<font color="#00007F">Close</font>
  <font color="#00007F">End</font> <font color="#00007F">With</font>

<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Function</font> AddToDatabase(sRecordSet <font color="#00007F">As</font> <font color="#00007F">String</font>,_ 
  sFieldToAdd <font color="#00007F">As</font> <font color="#00007F">String</font>, sValueToUpdate <font color="#00007F">As</font> <font color="#00007F">String</font>)

  <font color="#007F00">'Open the Contact table</font>
  <font color="#00007F">Set</font> RS = db.OpenRecordset(&quot;SELECT * FROM&quot; &amp; sRecordSet _ 
    &amp; &quot; Contact&quot;, dbOpenDynaset)

  <font color="#00007F">With</font> RS

    <font color="#007F00">'Set it to <font color="#00007F">Add</font> mode</font>
    .AddNew

    <font color="#007F00">'Enter the field values</font>
    .Fields(sFieldToUpdate).Value = sValueToUpdate

    <font color="#007F00">'Update it</font>
    .UpDate

    <font color="#007F00">'Close it</font>
    .<font color="#00007F">Close</font>
  <font color="#00007F">End</font> <font color="#00007F">With</font>

<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Function</font> DeleteFromDatabase(sRecordSet <font color="#00007F">As</font> <font color="#00007F">String</font>, _ 
  sField <font color="#00007F">As</font> <font color="#00007F">String</font>, sValue <font color="#00007F">As</font> <font color="#00007F">String</font>)

  <font color="#007F00">'Open the Contact table</font>
  <font color="#00007F">Set</font> RS = db.OpenRecordset(&quot;SELECT * FROM&quot;_ 
    &amp; sRecordSet &amp; &quot; WHERE&quot; &amp; sField &amp; &quot; = &quot; &amp; sValue &amp; &quot;'&quot;, dbOpenDynaset)

  <font color="#00007F">With</font> RS

    <font color="#007F00">'Set it to delete mode</font>
    .Delete

    <font color="#007F00">'Close it</font>
    .<font color="#00007F">Close</font>
  <font color="#00007F">End</font> <font color="#00007F">With</font>

<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Function</font> RepairDatabase(sDBName <font color="#00007F">As</font> <font color="#00007F">String</font>)
  db.<font color="#00007F">Close</font>

  DBEngine.RepairDatabase sDBName

  OpenDB (App.Path &amp; &quot;\demo.mdb&quot;)
<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Sub</font> CompactDatabase(sDBName <font color="#00007F">As</font> <font color="#00007F">String</font>, sBackup <font color="#00007F">As</font> <font color="#00007F">String</font>)

  db.<font color="#00007F">Close</font>

  <font color="#007F00">'Check to see if the tempory backup exists</font>
  <font color="#00007F">If</font> <font color="#00007F">Len</font>(<font color="#00007F">Dir</font>$(sBackup)) &gt; 0 <font color="#00007F">Then</font> <font color="#007F00">'Backup exists</font>

    <font color="#00007F">Kill</font> sBackup <font color="#007F00">'Delete it</font>

  <font color="#00007F">End</font> <font color="#00007F">If</font>

  <font color="#007F00">'Compact the database to the tempory one</font>
  DBEngine.CompactDatabase sDBName, sBackup

  <font color="#007F00">'Delete the current one</font>
  <font color="#00007F">Kill</font> sDBName

  <font color="#007F00">'Compact the backup to the actual database</font>
  DBEngine.CompactDatabase sBackup, sDBName

  <font color="#007F00">'And remove the backup database</font>
  <font color="#00007F">Kill</font> sBackup

  OpenDB (App.Path &amp; &quot;\demo.mdb&quot;)

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Sub</font> BackupDatabase(sDBPath <font color="#00007F">As</font> <font color="#00007F">String</font>, sOutput <font color="#00007F">As</font> <font color="#00007F">String</font>)

  <font color="#00007F">If</font> <font color="#00007F">Len</font>(<font color="#00007F">Dir</font>$(sDBPath)) &gt; 0 <font color="#00007F">Then</font> <font color="#007F00">'Backup exists</font>
    <font color="#00007F">Kill</font> sDBPath <font color="#007F00">'Delete it</font>
  <font color="#00007F">End</font> <font color="#00007F">If</font>

  <font color="#007F00">'Output to new database</font>
  DBEngine.CompactDatabase sOutput, sDBName

<font color="#00007F">End</font> <font color="#00007F">Sub</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica">

<p>Add a new form (frmTest). To the form add six command
buttons (cmdAdd, cmdDelete, cmdCompact, cmdUpdate, cmdNext and cmdRepair.) Add three text
boxes and a label (txtName, txtPhone, txtNotes and lblID). Copy the following code into
the form's General Declarations procedure:

</font><pre>
<font color="#007F00">'Private declarations for the database and recordset</font>
<font color="#00007F">Private</font> db <font color="#00007F">As</font> Database
<font color="#00007F">Private</font> RS <font color="#00007F">As</font> Recordset
<font color="#00007F">Private</font> m_cData <font color="#00007F">As</font> cDataFunc

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdAdd_Click()
  <font color="#007F00">'Set the global variables</font>
  sName = txtName
  sPhone = txtPhone
  sNotes = txtNotes

  <font color="#007F00">'Call the add function</font>
  m_cData.AddToDatabase &quot;Contact&quot;, &quot;Name&quot;, sName
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdCompact_Click()
  m_cData.CompactDatabase App.Path &amp; &quot;\demo.mdb&quot;, App.Path &amp; &quot;\backup.mdb&quot;
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdDelete_Click()
  <font color="#007F00">'Set the global variables</font>
  sName = txtNamer
  sPhone = txtPhone
  sNotes = txtNotes

  <font color="#007F00">'Call the delete function</font>
  m_cData.DeleteFromDatabase &quot;Contact&quot;, &quot;Name&quot;, sName
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdNext_Click()
  <font color="#007F00">'Call the next record function</font>
  m_cData.GetNextRecord &quot;Contact&quot;, &quot;Name&quot;, iID
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdRepair_Click()
  m_cData.RepairDatabase App.Path &amp; &quot;\demo.mdb&quot;
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> cmdUpdate_Click()
  <font color="#007F00">'Set the global variables</font>
  sName = txtName
  sPhone = txtPhone
  sNotes = txtNotes
  iID = lblID

  <font color="#007F00">'Call the Edit function</font>
  <font color="#00007F">Call</font> m_cData.Update(&quot;Contact&quot;, &quot;PersonID&quot;, lblID)
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Form_Load()
  <font color="#00007F">Set</font> m_cData = <font color="#00007F">New</font> cDataFunc
  m_cData.OpenDB (App.Path &amp; &quot;\demo.mdb&quot;)
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Form_Unload(Cancel <font color="#00007F">As</font> <font color="#00007F">Integer</font>)
  <font color="#00007F">Set</font> m_cData = Nothing
<font color="#00007F">End</font> <font color="#00007F">Sub</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica">
