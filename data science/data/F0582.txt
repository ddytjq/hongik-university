		function validate()
		{
			// validation of data entered
							
			var msg
							
			//check mandatory field

			if 
			( 
				(document.forms['form2'].txtAttendee1.value == "") ||	 
				(document.forms['form2'].txtPhoneNo.value == "") ||	 
				(document.forms['form2'].txtEmail.value == "") ||	 
				(document.forms['form2'].txtAddress.value == "") 
			) 
			{
				//build up error message
				if (document.forms['form2'].txtAttendee1.value == "")
				{
					msg = "Name";
				}
				if (document.forms['form2'].txtPhoneNo.value == "")
				{
					if (msg != null)
						{
							msg = msg + ", Phone Number";
						}
					else
						{
							msg = "Phone Number";
						}
				}
				if (document.forms['form2'].txtEmail.value == "")
				{
					if (msg != null)
						{
							msg = msg + ", Email Address";
						}
					else
						{
							msg = "Email Address"
						}
				}
				if (document.forms['form2'].txtAddress.value == "")
				{
					if (msg != null)
						{
							msg = msg + ", Address";
						}
					else
						{
							msg = "Address";
						}
				}

				
				msg = "Please enter your " + msg;
				alert (msg);
				return;

			
			}

			//Check that email address has been entered correctly
			
			var atsign = /@/i;
			var text = document.forms['form2'].txtEmail.value;
			var result = text.search(atsign);
			var index = text.length-1
			
			if (result == -1 || result == index)
			{
				msg = "Please enter valid email address";
				alert (msg);
				return;

			}

			//Choose at least some options!
			if 
			( 
				(document.forms['form2'].optOption1.checked == false) &&
				(document.forms['form2'].optOption2.checked == false) &&
				(document.forms['form2'].optOption3.checked == false) &&
				(document.forms['form2'].optOption4.checked == false) &&
				(document.forms['form2'].optOption5.checked == false) 			) 
			{
				msg = "Please choose a Conference Option first";
				alert (msg);
				return;
			}
			
			//if you choose to attend a workshop you must choose which one
			
			if (document.forms['form2'].optOption2.checked == true)
			{
				//member chosen to attend workshop
				
				if (
				(document.forms['form2'].MemWS1[0].checked == false) &&
				(document.forms['form2'].MemWS1[1].checked == false) &&
				(document.forms['form2'].MemWS1[2].checked == false)
				)

				{
					//but hasn't chosen which one
					msg = "Please choose which workshop you wish to attend";
					alert (msg);
					return;
						
				}
			}			
			else
			{
				if (document.forms['form2'].optOption4.checked == true)
				{
					//nonmember chosen to attend workshop
					
					if (
					(document.forms['form2'].NonMemWS1[0].checked == false) &&
					(document.forms['form2'].NonMemWS1[1].checked == false) &&
					(document.forms['form2'].NonMemWS1[2].checked == false)
					)

					{
						//but hasn't chosen which one
						msg = "Please choose which workshop you wish to attend";
						alert (msg);
						return;
							
					}
				}			
			}
			
			//make sure user doesn't enter a membership num and request to join vbug
			
			if 
			( 
				(document.forms['form2'].txtMembershipNo.value != "") &&	 
				(document.forms['form2'].optOption5.checked == true)
			)

				//both membership num and request to join vbug specified
				msg = "Please enter your membership number OR request to join as a member";
				alert (msg);
				return;
							
			}


			//if you choose any member options you must give membership number or choose to join vbug

				(document.forms['form2'].optOption1.checked == true) ||
				(document.forms['form2'].optOption2.checked == true)
			)

				//member options chosen
				if

					msg = "You've specified options which are for members only. Please enter your membership number or join as a member to enjoy these prices";

					return;
						
				}
