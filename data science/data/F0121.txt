    <font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">GoTo</font> ErrorHandler:

    PromptOnErr = PromptOnError

    <font color="#007F00">'// Check that AppPath exists.</font>
    <font color="#00007F">If</font> <font color="#00007F">Dir</font>(AppPath, vbNormal) = "" <font color="#00007F">Then</font>
        <font color="#00007F">If</font> PromptOnError <font color="#00007F">Then</font> MsgBox "The application path <font color="#007F00">'" & _ </font>

            vbCritical + vbOKOnly, "DLL/OCX Register"

        CreateFileAss = False
        <font color="#00007F">Exit</font> <font color="#00007F">Function</font>
    <font color="#00007F">End</font> <font color="#00007F">If</font>

    <font color="#00007F">Dim</font> ERROR_CHARS <font color="#00007F">As</font> String: ERROR_CHARS = "\/:*?<>|" & <font color="#00007F">Chr</font>(34)
    <font color="#00007F">Dim</font> i <font color="#00007F">As</font> <font color="#00007F">Integer</font>

    <font color="#00007F">If</font> <font color="#00007F">Asc</font>(Extension) <> 46 <font color="#00007F">Then</font> Extension = "." & Extension

    <font color="#007F00">'// Check for invalid chars within extension</font>
    <font color="#00007F">For</font> i = 1 <font color="#00007F">To</font> <font color="#00007F">Len</font>(Extension)
        <font color="#00007F">If</font> <font color="#00007F">InStr</font>(1, ERROR_CHARS, <font color="#00007F">Mid</font>(Extension, i, 1), vbTextCompare) <font color="#00007F">Then</font>
            <font color="#00007F">If</font> PromptOnError <font color="#00007F">Then</font> MsgBox "The file extension <font color="#007F00">'" _ </font>
                  & Extension & "' contains an illegal char (\/:*?<>|" _ 
                  & <font color="#00007F">Chr</font>(34) & ").", vbCritical + vbOKOnly, "DLL/OCX Register"

            <font color="#00007F">Exit</font> <font color="#00007F">Function</font>

    <font color="#00007F">Call</font> CreateSubKey(HKEY_CLASSES_ROOT, Action)           <font color="#007F00">'// Create action key</font>

            <font color="#007F00">'// This line of code sets the application's own icon as the _</font>

               FileType & "\DefaultIcon", <font color="#00007F">Trim</font>(AppPath & ",0"))

        <font color="#00007F">Else</font>
            <font color="#00007F">Call</font> SetKeyDefault(HKEY_CLASSES_ROOT, FileType _

        <font color="#00007F">End</font> <font color="#00007F">If</font>
    <font color="#00007F">End</font> I</p>

        CreateSubKey = False
    <font color="#00007F">Else</font>
        CreateSubKey = True

    <font color="#007F00">'// This function sets the default vaule of the key which is always a string</font>
    <font color="#00007F">Dim</font> regReply <font color="#00007F">As</font> <font color="#00007F">Long</font>, hKey <font color="#00007F">As</font> <font color="#00007F">Long</font>
    regReply = RegOpenKeyEx(RootKey, Address, 0, KEY_ALL_ACCESS, hKey)

    <font color="#00007F">If</font> regReply <> ERROR_SUCCESS <font color="#00007F">Then</font>
        <font color="#00007F">If</font> PromptOnErr <font color="#00007F">Then</font> MsgBox "An error occured while attempting _ 
          to access the registery.", vbCritical + vbOKOnly, "DLL/OCX Register"
        SetKeyDefault = False
        <font color="#00007F">Exit</font> <font color="#00007F">Function</font>
    <font color="#00007F">End</font> <font color="#00007F">If</font>

    Value = Value & <font color="#00007F">Chr</font>(0)

    regReply = RegSetValueExString(hKey, "", 0&, REG_SZ, Value, <font color="#00007F">Len</font>(Value))

    <font color="#00007F">If</font> regReply <> ERROR_SUCCESS <font color="#00007F">Then</font>
        <font color="#00007F">If</font> PromptOnErr <font color="#00007F">Then</font> MsgBox "An error occured while _

           vbOKOnly, "DLL/OCX Register"

        SetKeyDefault = False
    <font color="#00007F">Else</font>
