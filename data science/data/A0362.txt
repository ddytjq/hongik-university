			function getAvailableHeight()
				{
				var nav = "";

				var available_height;

					else if(parseInt(navigator.appVersion) == 3) 
						nav = "ns3";
					else if(parseInt(navigator.appVersion) == 2)

				if(nav == 'ns4')
