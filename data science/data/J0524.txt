// TB 06/19/00
function validate(){
var error_str='';
var error_focus_set=0;

var str=document.abuse_form.email;

error_str+='\nPlease include your e-mail address.';
if (!error_focus_set) error_focus_set=str;
}
else{
if (str.value.indexOf("@")<1){
error_str+='\nPlease enter a valid e-mail address.';
if (!error_focus_set) error_focus_set=str;
}
}

// confirm offending site URL is present
var str=document.abuse_form.url;
if (str.value.length<1){
error_str+='\nPlease include the site URL.';
if (!error_focus_set) error_focus_set=str;
}
else{
if (((str.value.charAt(0)=="h")||(str.value.charAt(0)=="H"))&&
((str.value.charAt(1)=="t")||(str.value.charAt(1)=="T"))&&
((str.value.charAt(2)=="t")||(str.value.charAt(2)=="T"))&&
((str.value.charAt(3)=="p")||(str.value.charAt(3)=="P"))&&
(str.value.charAt(4)==":")&&
(str.value.charAt(5)=="/")&&
(str.value.charAt(6)=="/")){
var temp=''+str.value.slice(7,str.value.length);
str.value=temp;
}
}

// validate option is selected
var str=document.abuse_form.reason;
for (var index=0; index<str.length; index++){
var test=str[index].checked;
if (test==true) break;
}
if (!test){
error_str+='\nPlease select a reason.';
if (!error_focus_set) error_focus_set=str[0];
}

// submit or display errors
if (error_str.length>0){
alert(error_str);

else{
document.abuse_form.abuse_submit.value="Next >>";
adjust_oneliner();
