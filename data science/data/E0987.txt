Overview: This article explains how to create a Direct Draw Sprite. The
sample draws a sprite in Window Mode. This is code commenting out for
full screen mode.
<a href="http://www.listensoftware.com/ddsprite.zip">Download Sample</a>
<b>What are Sprites?</b>
1. Sprites are Bitmap resources that can be drawn to a
full screen or window mode. Sprites a 2D images that
can be moved around the screen. Frames are viewports
into the bitmap resources giving access to multiple
images from one bitmap resource.
All Direct Draw programs require at least three components: the Direct Draw object, 
a front surface, and a back surface. All rendering is done to the back surface, 
which is then flipped with the front surface at vertical retrace. Double
buffering creates smooth animation.
Setup header file and libraries
#include &lt;ddraw.h&gt;
#pragma comment ( lib, "ddraw.lib" ) 
#pragma comment ( lib, "dxguid.lib" )
2. In this example the Sprite Structure controls the
location of the bitmap and the Frame Index.
	struct SPRITE_STRUCT

    	{
    		LONG lFrame;
    		FLOAT fPosX; 
    		FLOAT fPosY;
    	};
    3. The frame index is a offset into
    the Bitmap resource, IDB_ANIMATE_SHEET which was imported in from a 
    BMP file. Each Frame is 48 bits wide. The bitmap file dimensions
    are five frames width by six frames height. The total number of
    frames is thirty.
    <B>Data Structure</B>
    <pre>
    #define SPRITE_DIAMETER 48
    #define NUM_SPRITES 1
    #define NUM_FRAMES 30
    </pre>
    4. Initialize DirectDraw and the Frames
    	a. Set the Direct Draw device to the current window
    	b. Use the Bitmap resource to create the pallette
    	c. Create a DirectDraw Surface. Think of a surface as
    	access to the video memory. Images put to the the
    	surface are rendered.
    	d. Create a Surface for the 2D Text Message
    	e. Define Frames which are RECT contain offsets into the Bitmap surface
    	[top,left,bottom,right]
    <font color=purple>
    HRESULT CDdspriteView::InitDirectDraw()

        {
        HRESULT hr;
        	LPDIRECTDRAWPALETTE pDDPal = NULL; 
        g_pDisplay = new CDisplay();
        if( FAILED( hr = g_pDisplay->CreateWindowedDisplay( AfxGetMainWnd()->m_hWnd, WINDOW_WIDTH, WINDOW_HEIGHT ) ) )

            {
            AfxMessageBox( TEXT("Failed initializing DirectDraw."));
            return hr;
        }
<br>        <font color=#186125>// Create and set the palette when in pa
</font>        <font color=#186125>//     lettized color
</font>        if( FAILED( hr = g_pDisplay->CreatePaletteFromBitmap( &pDDPal, MAKEINTRESOURCE( IDB_ANIMATE_SHEET ) ) ) )
        return hr;
        g_pDisplay->SetPalette( pDDPal );
        SAFE_RELEASE( pDDPal );
        <font color=#186125>// Create a surface, and draw a bitmap r
</font>        <font color=#186125>//     esource on it. 
</font>        if( FAILED( hr = g_pDisplay->CreateSurfaceFromBitmap( &g_pAnimationSurface, 
        MAKEINTRESOURCE( IDB_ANIMATE_SHEET ), 
        SPRITE_DIAMETER * 5, 
        SPRITE_DIAMETER * 6 ) ) )
        return hr;
        <font color=#186125>// Create a surface, and draw text to it
</font>        <font color=#186125>//     . 
</font>        if( FAILED( hr = g_pDisplay->CreateSurfaceFromText( &g_pTextSurface, NULL, HELPTEXT, 
        RGB(0,0,0), RGB(255, 255, 0) ) ) )
        return hr;
        <font color=#186125>// Set the color key for the logo sprite
</font>        <font color=#186125>//     to black
</font>        if( FAILED( hr = g_pAnimationSurface->SetColorKey( 0 ) ) )
        return hr;
        	g_Sprite.fPosX = WINDOW_WIDTH/2.0f;
        	g_Sprite.fPosY = WINDOW_HEIGHT/2.0f; 
        	g_Sprite.lFrame = 0;
        <font color=#186125>// The source rects are used during the 
</font>        <font color=#186125>//     blt of the dds to the backbuffer. 
</font>        for( int iFrame = 0; iFrame < NUM_FRAMES; iFrame++ )

            {
            g_rcFrame[iFrame].top= (iFrame / 5) * SPRITE_DIAMETER;
            g_rcFrame[iFrame].left= (iFrame % 5) * SPRITE_DIAMETER;
            g_rcFrame[iFrame].bottom = g_rcFrame[iFrame].top + SPRITE_DIAMETER;
            g_rcFrame[iFrame].right = g_rcFrame[iFrame].left + SPRITE_DIAMETER;
        }
<br>        return S_OK;
    }
<br>    </font>
    5. Draw the bitmap
    a. Fill the back buffer with black, ignoring errors until the flip
    b. Blt the help text on the backbuffer
    c. Draw the sprite onto the back buffer
    <font color=purple>
    HRESULT CDdspriteView::DisplayFrame()

        {
        	HRESULT hr;
        <font color=#186125>// Fill the back buffer with black, igno
</font>        <font color=#186125>//     ring errors until the flip
</font>        g_pDisplay->Clear( 0 );
        <font color=#186125>// Blt the help text on the backbuffer, 
</font>        <font color=#186125>//     ignoring errors until the flip
</font>        g_pDisplay->Blt( 10, 10, g_pTextSurface, NULL );
        <font color=#186125>// Blt all the sprites onto the back buf
</font>        <font color=#186125>//     fer using color keying,
</font>        <font color=#186125>// ignoring errors until the flip. Note 
</font>        <font color=#186125>//     that all of these sprites 
</font>        <font color=#186125>// use the same DirectDraw surface.
</font>        g_pDisplay->Blt( (DWORD)g_Sprite.fPosX, 
        (DWORD)g_Sprite.fPosY, 
        g_pAnimationSurface, 
        &g_rcFrame[g_Sprite.lFrame]);
        <font color=#186125>// We are in fullscreen mode, so perform
</font>        <font color=#186125>//     a flip and return 
</font>        <font color=#186125>// any errors like DDERR_SURFACELOST
</font>        if( FAILED( hr = g_pDisplay->Present() ) )
        return hr;
        return S_OK;
    }
<br>    </font>
    6. User Interaction
    a. The Arrows keys reposition the bitmap
    b. The 1 and 2 keys cycle through the frame by index

                                    			{
