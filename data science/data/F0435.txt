<p><font face="Arial, Helvetica" size="+1"><b>Make a resizing control</b></font></p>
<p><i>By <a href="/php-bin/authorbio.php?authorid=31&associd=89&assoctype=2">Sam Huggill</a></i></p>
<p><font face="Arial, Helvetica" size="+1"><b>vbsResize.ocx</b></font></p>

<p>Add a usercontrol (vbsResize) and copy the following code into it:</p>   

</font><pre><font color="#00007F">Option</font> <font color="#00007F">Explicit</font>

<font color="#007F00">'// Are we sizing font?</font>
<font color="#00007F">Private</font> m_blnSizeFont <font color="#00007F">As</font> <font color="#00007F">Boolean</font>

<font color="#007F00">'// Should we keep the ratio?</font>
<font color="#00007F">Private</font> m_blnRatio <font color="#00007F">As</font> <font color="#00007F">Boolean</font>

<font color="#007F00">'// Stores info on each ctrl</font>
<font color="#00007F">Private</font> <font color="#00007F">Type</font> TCtlInfo
    ctrl <font color="#00007F">As</font> Control
    <font color="#00007F">Left</font> <font color="#00007F">As</font> <font color="#00007F">Single</font>
    Top <font color="#00007F">As</font> <font color="#00007F">Single</font>
    Width <font color="#00007F">As</font> <font color="#00007F">Single</font>
    Height <font color="#00007F">As</font> <font color="#00007F">Single</font>
    FontSize <font color="#00007F">As</font> <font color="#00007F">Integer</font>
<font color="#00007F">End</font> <font color="#00007F">Type</font>

<font color="#007F00">'// Controls array</font>
<font color="#00007F">Dim</font> Controls() <font color="#00007F">As</font> TCtlInfo

<font color="#007F00">'// Our parent form</font>
<font color="#00007F">Private</font> WithEvents ParentForm <font color="#00007F">As</font> Form

<font color="#007F00">'// Parent Form's Width and Height</font>
<font color="#00007F">Private</font> m_sngPWidth <font color="#00007F">As</font> <font color="#00007F">Single</font>
<font color="#00007F">Private</font> m_sngPHeight <font color="#00007F">As</font> <font color="#00007F">Single</font>

<font color="#007F00">'// Original ratio of height/width</font>
<font color="#00007F">Private</font> m_sngHWRatio <font color="#00007F">As</font> <font color="#00007F">Single</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> ParentForm_Load()
    m_sngPWidth = 0
    <font color="#007F00">'// Save the original ratio </font>
    m_sngHWRatio = ParentForm.Height / ParentForm.Width
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> ParentForm_Resize()
    <font color="#00007F">If</font> m_sngPWidth = 0 <font color="#00007F">Then</font>
        Rebuild
    <font color="#00007F">Else</font>
        Refresh
    <font color="#00007F">End</font> <font color="#00007F">If</font>
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> UserControl_ReadProperties(PropBag <font color="#00007F">As</font> PropertyBag)
    <font color="#00007F">If</font> Ambient.UserMode = False <font color="#00007F">Then</font> <font color="#00007F">Exit</font> <font color="#00007F">Sub</font>
    <font color="#007F00">'// Store the parent form </font>
    <font color="#00007F">Set</font> ParentForm = Parent
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Rebuild()
    <font color="#00007F">Dim</font> intCount <font color="#00007F">As</font> <font color="#00007F">Integer</font>
    <font color="#00007F">Dim</font> ctrl <font color="#00007F">As</font> Control
    
    <font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">Resume</font> <font color="#00007F">Next</font>
    
    <font color="#00007F">If</font> Ambient.UserMode = False <font color="#00007F">Then</font> <font color="#00007F">Exit</font> <font color="#00007F">Sub</font>
    <font color="#007F00">'// Make a reference to the parent </font>
    <font color="#00007F">Set</font> ParentForm = UserControl.Parent
    <font color="#007F00">'// Store the height and width </font>
    m_sngPWidth = ParentForm.ScaleWidth
    m_sngPHeight = ParentForm.ScaleHeight
    
    <font color="#00007F">ReDim</font> Controls(ParentForm.Controls.Count - 1) <font color="#00007F">As</font> TCtlInfo
    <font color="#007F00">'// Store the sizes of all the controls </font>
    <font color="#00007F">For</font> intCount = 0 <font color="#00007F">To</font> ParentForm.Controls.Count - 1
        <font color="#00007F">Set</font> ctrl = ParentForm.Controls(intCount)
        <font color="#00007F">With</font> Controls(intCount)
            <font color="#00007F">Set</font> .ctrl = ctrl
            .<font color="#00007F">Left</font> = ctrl.<font color="#00007F">Left</font>
            .Top = ctrl.Top
            .Width = ctrl.Width
            .Height = ctrl.Height
            .FontSize = ctrl.Font.Size
        <font color="#00007F">End</font> <font color="#00007F">With</font>
    <font color="#00007F">Next</font>
<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Public</font> <font color="#00007F">Sub</font> Refresh()
    <font color="#00007F">Dim</font> intCount <font color="#00007F">As</font> <font color="#00007F">Integer</font>
    <font color="#00007F">Dim</font> ctrl <font color="#00007F">As</font> Control
    <font color="#00007F">Dim</font> sngWFactor <font color="#00007F">As</font> <font color="#00007F">Single</font>
    <font color="#00007F">Dim</font> sngHFactor <font color="#00007F">As</font> <font color="#00007F">Single</font>
    <font color="#00007F">Dim</font> sngMinFactor <font color="#00007F">As</font> <font color="#00007F">Single</font>
    
    <font color="#00007F">Static</font> blnExec <font color="#00007F">As</font> <font color="#00007F">Boolean</font>
    <font color="#00007F">If</font> blnExec <font color="#00007F">Then</font> <font color="#00007F">Exit</font> <font color="#00007F">Sub</font>
    
    <font color="#00007F">If</font> Ambient.UserMode = False <font color="#00007F">Then</font> <font color="#00007F">Exit</font> <font color="#00007F">Sub</font>
    
    <font color="#00007F">If</font> m_blnRatio <font color="#00007F">Then</font>
        blnExec = True
         
         <font color="#007F00">'// Keep the original ratio </font>
        ParentForm.Height = m_sngHWRatio * ParentForm.Width
        blnExec = False
    <font color="#00007F">End</font> <font color="#00007F">If</font>
    
    <font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">Resume</font> <font color="#00007F">Next</font>
    
    sngWFactor = ParentForm.ScaleWidth / m_sngPWidth
    sngHFactor = ParentForm.ScaleHeight / m_sngPHeight
    <font color="#007F00">'// Take the smallest one </font>
    <font color="#00007F">If</font> sngWFactor &lt; sngHFactor <font color="#00007F">Then</font> 
        sngMinFactor= sngWFactor 
    <font color="#00007F">Else</font> 
        sngMinFactor= sngHFactor 
    <font color="#00007F">End</font> <font color="#00007F">If</font> 
    <font color="#007F00">'// Iterate through the controls</font>
    <font color="#007F00">'// changing their sizes</font>
     <font color="#00007F">For</font> intCount = 0
<font color="#00007F">To</font> <font color="#00007F">UBound</font>(Controls)
        <font color="#00007F">With</font> Controls(intCount)
 
            <font color="#007F00">'// We must resize the font before</font>
            <font color="#007F00">'// the control is resized to account</font>
            <font color="#007F00">'// for scrollbars</font>
            
<font color="#00007F">If</font> m_blnSizeFont <font color="#00007F">Then</font>             
.ctrl.Font.Size = .FontSize * sngMinFactor
<font color="#00007F">End</font> <font color="#00007F">If</font>
    .ctrl.<font color="#00007F">Left</font> = .<font color="#00007F">Left</font> * sngWFactor
    .ctrl.Top = .Top * sngHFactor
    .ctrl.Width = .Width * sngWFactor
    .ctrl.Height = .Height * sngHFactor
<font color="#00007F">End</font> <font color="#00007F">With</font>

<font color="#00007F">Next</font>

<font color="#00007F">End</font> Sub</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica">


</font>
