<p><font face="Arial, Helvetica" size="+1"><b>Placing a progress bar in a status bar panel</b></font></p>
<p><i>Tip by <a href="/php-bin/authorbio.php?authorid=31&associd=339&assoctype=3">Sam Huggill</a></i></p>

<table border="0" width="100%" bgcolor="#FFCC00">
  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip description</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="Arial" size="2"></font></td>  </tr>
</table><br>
<table border="0" width="100%" bgcolor="#FFCC00">

<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="arial" size="-1" ><p>This tip was written by Chris Eastwood (www.codeguru.com/vb/) To use this tip add a form with a progress bar, a status bar and a command button on. Copy the following code into the form's General Declarations procedure:</p>

</font><pre>'Declarations
<font color="#007F00">'</font>
<font color="#007F00">' API Declarations</font>
<font color="#007F00">'</font>
<font color="#00007F">Private</font> <font color="#00007F">Declare</font> <font color="#00007F">Function</font> SetParent <font color="#00007F">Lib</font> "user32" (<font color="#00007F">ByVal</font> hWndChild <font color="#00007F">As</font> _
<font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> hWndNewParent <font color="#00007F">As</font> <font color="#00007F">Long</font>) <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Private</font> <font color="#00007F">Declare</font> <font color="#00007F">Function</font> SendMessageAny <font color="#00007F">Lib</font> "user32" <font color="#00007F">Alias</font> "SendMessageA" _
(<font color="#00007F">ByVal</font> hwnd <font color="#00007F">As</font> <font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> msg <font color="#00007F">As</font> <font color="#00007F">Long</font>, <font color="#00007F">ByVal</font> wParam <font color="#00007F">As</font> <font color="#00007F">Long</font>, lParam _
<font color="#00007F">As</font> <font color="#00007F">Any</font>) <font color="#00007F">As</font> <font color="#00007F">Long</font>

<font color="#007F00">'</font>
<font color="#007F00">' API Types</font>
<font color="#007F00">'</font>
<font color="#007F00">' RECT is used to get the size of the panel we're inserting into</font>
<font color="#007F00">'</font>
<font color="#00007F">Private</font> <font color="#00007F">Type</font> RECT
    <font color="#00007F">Left</font> <font color="#00007F">As</font> <font color="#00007F">Long</font>
    Top <font color="#00007F">As</font> <font color="#00007F">Long</font>
    <font color="#00007F">Right</font> <font color="#00007F">As</font> <font color="#00007F">Long</font>
    Bottom <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">End</font> <font color="#00007F">Type</font>

<font color="#007F00">'</font>
<font color="#007F00">' API Messages</font>
<font color="#007F00">'</font>
<font color="#00007F">Private</font> <font color="#00007F">Const</font> WM_USER <font color="#00007F">As</font> <font color="#00007F">Long</font> = &H400
<font color="#00007F">Private</font> <font color="#00007F">Const</font> SB_GETRECT <font color="#00007F">As</font> <font color="#00007F">Long</font> = (WM_USER + 10)


<font color="#00007F">Private</font> <font color="#00007F">Sub</font> ShowProgressInStatusBar(<font color="#00007F">ByVal</font> bShowProgressBar <font color="#00007F">As</font> <font color="#00007F">Boolean</font>) 

    <font color="#00007F">Dim</font> tRC <font color="#00007F">As</font> RECT

    <font color="#00007F">If</font> bShowProgressBar <font color="#00007F">Then</font>
        <font color="#007F00">'</font>
        <font color="#007F00">' <font color="#00007F">Get</font> the size of the Panel (2) Rectangle from the status bar</font>
        <font color="#007F00">' remember that Indexes in the API are always 0 based (well,</font>
        <font color="#007F00">' nearly always) - therefore Panel(2) = Panel(1) to the api</font>
        <font color="#007F00">'</font>
        <font color="#007F00">'</font>
        SendMessageAny StatusBar1.hwnd, SB_GETRECT, 1, tRC
        <font color="#007F00">'</font>
        <font color="#007F00">' and convert it to twips....</font>
        <font color="#007F00">'</font>
        <font color="#00007F">With</font> tRC
            .Top = (.Top * Screen.TwipsPerPixelY)
            .<font color="#00007F">Left</font> = (.<font color="#00007F">Left</font> * Screen.TwipsPerPixelX)
            .Bottom = (.Bottom * Screen.TwipsPerPixelY) - .Top
            .<font color="#00007F">Right</font> = (.<font color="#00007F">Right</font> * Screen.TwipsPerPixelX) - .<font color="#00007F">Left</font>
        <font color="#00007F">End</font> <font color="#00007F">With</font>
        <font color="#007F00">'</font>
        <font color="#007F00">' <font color="#00007F">Now</font> Reparent the ProgressBar to the statusbar</font>
        <font color="#007F00">'</font>
        <font color="#00007F">With</font> ProgressBar1
            SetParent .hwnd, StatusBar1.hwnd
            .Move tRC.<font color="#00007F">Left</font>, tRC.Top, tRC.<font color="#00007F">Right</font>, tRC.Bottom
            .Visible = True
            .Value = 0
        <font color="#00007F">End</font> <font color="#00007F">With</font>

    <font color="#00007F">Else</font>
        <font color="#007F00">'</font>
        <font color="#007F00">' Reparent the progress bar back to the form and hide it</font>
        <font color="#007F00">'</font>
        SetParent ProgressBar1.hwnd, <font color="#00007F">Me</font>.hwnd
        ProgressBar1.Visible = False
    <font color="#00007F">End</font> <font color="#00007F">If</font>

<font color="#00007F">End</font> <font color="#00007F">Sub</font>

<font color="#00007F">Private</font> <font color="#00007F">Sub</font> Command1_Click()
    <font color="#007F00">'Call the function</font>
    <font color="#00007F">Call</font> ShowProgressInStatusBar(True)
    
    <font color="#007F00">'Do something in the progress bar</font>
    <font color="#00007F">For</font> i = 1 <font color="#00007F">To</font> ProgressBar1.Max
        ProgressBar1.Value = i
    <font color="#00007F">Next</font> i

<font color="#00007F">End</font> <font color="#00007F">Sub</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica"></font></td></tr></table>
