function check_numeric(val, name)
{ 
var decpnt=0

for (i=0; i<val.length; i++)
{ ch=val.charAt(i)
if ((ch<'0' || ch>'9')&&(ch!='.'))
{ alert(name+' must be numeric')
return false
}
else
{ if (ch=='.')
if (decpnt==0)
decpnt++
else
{ alert(name+' must be numeric')
return false
}
}
}
return true
}
function format_money(amt)
{
amt=''+amt

if (amt.length==0)
return '0.00'

decidx=amt.indexOf('.')

if (decidx==0)
{ amt='0'+amt
decidx++
}
if (decidx==-1)
return amt+'.00'

if (amt.length-decidx==2)
return amt+'0'

return amt.substring(0, decidx+3);
}
