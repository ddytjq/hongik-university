function validateDOB(dateIn,monthIn,yearIn){


//  ----------- Start of Include file ---------------
// For all form validation javascripts
// Used to define browser dependent document form object

	if ((navigator.appName == "Netscape") && (navigator.userAgent.toLowerCase().indexOf('gecko') < 0)){
	
		formLocation = document.contentsLayer.document.forms[0];}  //Older Netscape browsers
	else{
		formLocation = document.forms[0];}  // MS IE, Netscape v6, Gecko powered browsers

	
// ---------- Endo of Include file -----------------

	//create text string for comparision
	var textStr = dateIn+"/"+monthIn+"/"+yearIn

	//create new date type
	var chkDate=new Date()

	//set the date, month and year based on what was
	//passed in

	//now we have created the text string, strip any preceding 0s from
	//dateIn, so that jS don't have a fit!
	if(dateIn.indexOf("0") == 0){
		dateIn = dateIn.substring(1)
	}
	//do the same for month
	if(monthIn.indexOf("0") == 0){
		monthIn = monthIn.substring(1)
	}


	chkDate.setMonth((parseInt(monthIn)-1),(parseInt(dateIn)))
	chkDate.setFullYear(parseInt(yearIn))

	//reverse this to get out what javascript is
	//the real date, plus put a 0 on the front of single date values
	var chkMonthInt  = chkDate.getMonth()+1
	var chkMonthStr = chkMonthInt+"/"
	if (chkMonthStr.length == 2){
		chkMonthStr = "0"+chkMonthStr
	}

	var chkDateInt = chkDate.getDate()
	var chkDateStr = chkDateInt+"/"
	if (chkDateStr.length == 2){
		chkDateStr = "0"+chkDateStr
	}

	var cmpDate=chkDateStr+chkMonthStr+(chkDate.getFullYear())

	//uncomment for debug
	//	alert("textStr = " + textStr + " cmpDate = " + cmpDate);
	if (parseInt(yearIn, 10)  < 1753) {
		alert("Date Of Birth Problem\nYou've entered a year earlier than 1753. Unfortunately we can only support dates after 1753, because this is when the Gregorian and Julian calendars were synchronized! Please re-enter a more recent date.")
	return false
	}
	//check the str length (must be 10 chrs)
	if (cmpDate.length != 10){
		alert("Date Of Birth Problem\nYou've entered an invalid date or date format.  Please use the dd/mm/yyyy format.");
		return false;
	}
	else {
		if (textStr!=cmpDate){
			alert("Date Of Birth Problem\nYou've entered an invalid date or date format.  Please use the dd/mm/yyyy format.");
			return false;
		}
		else {
			if (cmpDate=="NaN/NaN/NaN"){
				alert("Date Of Birth Problem\nYou've entered an invalid date or date format.  Please use the MM/DD/YY format.");
				return false;
			}
			else {
				return true;
			}
		}
	}
}

function validatePremiumForm(){


//  ----------- Start of Include file ---------------
// For all form validation javascripts
// Used to define browser dependent document form object

	if ((navigator.appName == "Netscape") && (navigator.userAgent.toLowerCase().indexOf('gecko') < 0)){
	
		formLocation = document.contentsLayer.document.forms[0];}  //Older Netscape browsers
	else{
		formLocation = document.forms[0];}  // MS IE, Netscape v6, Gecko powered browsers

	
// ---------- Endo of Include file -----------------
           
	//validate postcode
	if (!validate_postcode()){
		return false;
	}

	//validate dob
	if (!validate_dob()){
		return false;
	}
	//validate that a policy type has been chosen
	if ((!formLocation.contentsRequired.checked) && (!formLocation.buildingsRequired.checked)){
		alert("Insurance Type not Chosen\n\nYou must indicate whether you want contents insurance, buildings insurance, or both");
		formLocation.contentsRequired.focus();
		return false;
	}

	//validate occupation
	if (formLocation.occupation.options[formLocation.occupation.selectedIndex].value == 1){
		alert("Sorry - RapidInsure is unable provide insurance for this category of Occupation.")
		formLocation.occupation.focus();
		return false;
	}

	//validate occupation
	if (formLocation.occupation.options[formLocation.occupation.selectedIndex].value == 11){
		alert("Sorry - RapidInsure is unable provide insurance for this category of Occupation.")
		formLocation.occupation.focus();
		return false;
	}

	//validate occupation
	if (formLocation.occupation.options[formLocation.occupation.selectedIndex].value == 99){
		alert("Sorry - RapidInsure is unable provide insurance for this category of Occupation.")
		formLocation.occupation.focus();
		return false;
	}

	//if builidngs required
	if(formLocation.buildingsRequired.checked){
		//validate building against tennancy
		if (formLocation.ownership.options[formLocation.ownership.selectedIndex].value == "Tenant"){
			alert("Sorry - Rapidnsure is unable to provide buildings cover to tenants")
			formLocation.buildingsRequired.checked = false;
			formLocation.ownership.focus();
			return false;
		}

		//validate against flats
		if (formLocation.construction.options[formLocation.construction.selectedIndex].value == "Flat"){
			alert("Sorry - RapidInsure is unable to provide buildings insurance for flats")
			formLocation.buildingsRequired.checked = false;
			formLocation.construction.focus();
			return false;
		}

		//validate buildings cover required
		if (!validate_amountB()){
			return false;
		}

		//validate building type
		if (formLocation.material.options[formLocation.material.selectedIndex].value == "Other"){
			alert("Sorry - RapidInsure is unable to provide insurance if your property\nis not constructed of brick, stone, concrete or flint.")
			formLocation.material.focus();
			return false;
		}

		//validate roof type
		if (formLocation.roof.options[formLocation.roof.selectedIndex].value == "Other"){
			alert("Sorry - RapidInsure is unable to provide insurance if your property roof\nis not constructed of tile, slate, concrete or metal.")
			formLocation.material.focus();
			return false;
		}

	}
	return true;

}
