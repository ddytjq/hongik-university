function isEmpty(field){
    if(field==null) return true;
    if(field.length<1) return true;
    for(i=0;i<field.length;i++) {
        if(field.charAt(i)!=' ') return false;
    }
    return true;
}

function isNotIn(field, txt){
    if(field==null) return true;
    if(field.length<1) return true;
    for(i=0;i<field.length;i++) {
        if(field.charAt(i)==txt) {return false;}
    }
    return true;
}
