// This function switch on/off the image associated to a layer
// Function params: imgName=image to switch on
function fnClickImage(imgName)
{		

		x=MM_findObj(imgName);
		y=MM_findObj(OldValueClicked);
		
		if (OldValueClicked!=imgName)
		{
			imgOn = eval(imgName + "_on.src");
			x.src = imgOn;
			imgOff = eval(OldValueClicked + "_off.src");
			y.src = imgOff;
		}
		OldValueClicked=imgName;
}

// This function change the visibility layer and call fnClickImagePlayers function
// Function params: idTable=layer to switch on; imgName=image to switch on
function switchTable(idTable,imgName)
{
	switch (idTable)
	{
		case 1:
			MM_showHideLayers('tbl1Div','','show','tbl2Div','','hide','tbl3Div','','hide');
			break;
		case 2:
			MM_showHideLayers('tbl1Div','','hide','tbl2Div','','show','tbl3Div','','hide');
			break;
		case 3:
			MM_showHideLayers('tbl1Div','','hide','tbl2Div','','hide','tbl3Div','','show');
			break;			
	}
	fnClickImage(imgName);
	
	return false;
}
