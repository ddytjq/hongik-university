<p align="left">#include &lt;iostream&gt;&nbsp;&nbsp;&nbsp; // C++ input/output stream<br>
#include &lt;string>&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp; // string class<br>
using namespace std;&nbsp;&nbsp; // needed under Microsoft Visual C++ Compiler<br>
&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
// all other versions, leave out the namespace std;<br>
<br>
int main() {<br>
<br>
	string name;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;
// member of string class<br>
<br>
	// prompts user to enter their first name<br>
	cout &lt;&lt; "Please enter your first name (quit to stop): ";<br>
	cin >> name;<br>
<br>
	// Loop to repeatedly ask the users first name until End of File is met or<br>
	// the word quit or QUIT is entered<br>
	while (!cin.eof() &amp;&amp; name != "quit" &amp;&amp; name != "QUIT") {  // &amp;&amp; - and for C/C++<br>
		cout &lt;&lt; "Hello " &lt;&lt; name &lt;&lt; " nice to meet you!\n";   // || - or operator<br>
		cout &lt;&lt;"\nPlease enter first your name (quit to stop): ";<br>
		cin >> name;<br>
	}<br>
<br>
	return 0;     // exit program<br>
<br>
/* The syntax for a while loop is<br>
     while (test condition) {&nbsp;<br>
        ~statements go in here~&nbsp;<br>
	 }<br>
brackets are only necessary if the loop contains more than one command, the best bet is<br>
to always add brackets just in case you have to go back and add a statement later, that<br>
way your program won't suddenly contain 100 errors&nbsp;<br>
*/<br>
}</p>
