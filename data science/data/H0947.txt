function img_act(imgName)
{
     if (document.images)
          {
        
        // change image and identify page contents
    
       imgOn = eval(imgName + "on.src");
        stOn = eval(imgName + "bottom");
        window.status = stOn;   
        if(imgName=="esd"||imgName=="lsd"||imgName=="msd"){imgName="div"}
       
        document[imgName].src = imgOn;
        }
}

function img_inact(imgName)
{
        if (document)
        {
 
       // original image

     document[imgName].src = eval(imgName + "off.src");
        window.status = ""; 
      }
}
