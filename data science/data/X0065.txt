function FWFindImage(doc, name, j) {
    var theImage = false;
    if (doc.images) {
        theImage = doc.images[name];
    }
    if (theImage) {
        return theImage;
    }
    if (doc.layers) {
        for (j = 0; j < doc.layers.length; j++) {
            theImage = FWFindImage(doc.layers[j].document, name, 0);
            if (theImage) {
                return (theImage);
            }
        }
    }
    return (false);
}

            <div align="left"></div>
            <p> <img border="0" src="images/lines.gif" width="457" height="11"> 
            <p>A group of eight runners representing World Runners, headquartered 
              in the Bay Area, are embarking on an educational trip to Cuba and 
              will be participating in the Havana Marathon on November 18. The 
              trip is cosponsored and coordinated by Global Exchange, headquartered 
              in San Francisco. The group will be visiting Cuba under a Treasury 
              Department License from November 14 to 22. 
            <p><img src="events/cuba/countrysidemountain2.jpg" width="241" height="213">
            <p>This trip will introduce the runners to a culture rarely seen by 
              American resident. While there, the runners will be visiting schools 
              and farms and be participating in cultural activities. They are 
              also scheduled to visit social programs. 
            <p>Among the runners will be Emil Gluekler, the Managing Director 
              of World Runners. Emil has a goal to run in 100 marathons and raise 
              funding for children's health and education programs. Emil recently 
              participated in the Silicon Valley Marathon, his 30th marathon, 
              and has been averaging 2 marathons a year. <a href="events/cuba/index.htm"><br>
              [Click here for more information]</a>
