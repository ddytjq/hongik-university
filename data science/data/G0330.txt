function compute_weight(form)
{
    var weight = form.wt.value;
    if (weight > 0.0) {
        form.outputmrc.value = int_zero( 10 * weight * .378 ) / 10;
        form.outputvn.value = int_zero( 10 * weight * .906 ) / 10;
        form.outputmoon.value = int_zero( 10 * weight * .166 ) / 10;
        form.outputmars.value = int_zero( 10 * weight * .379 ) / 10;
        form.outputjp.value = int_zero( 10 * weight * 2.533 ) / 10;
        form.outputsat.value = int_zero( 10 * weight * 1.066 ) / 10;
        form.outputur.value = int_zero( 10 * weight * .905 ) / 10;
        form.outputnpt.value = int_zero( 10 * weight * 1.133 ) / 10;
        form.outputplt.value = int_zero( 10 * weight * .067 ) / 10;
        form.outputj2.value = int_zero( 10 * weight * .183 ) / 10;
        form.outputj3.value = int_zero( 10 * weight * .134 ) / 10;
        form.outputj4.value = int_zero( 10 * weight * .145 ) / 10;
        form.outputj5.value = int_zero( 10 * weight * .127 ) / 10;
        form.outputsun.value = int_zero( 10 * weight * 27.072 ) / 10;
        form.outputwd.value = int_zero( 10 * weight * 1.3E+6 ) / 10;
        form.outputns.value = weight * 1.4E+11;
    }

}

// Function to return 0 if result is <1
function int_zero(x)
{
	if ( x < 1 )
		return 0 ;
	else
		return parseInt( x ,10 );
}
