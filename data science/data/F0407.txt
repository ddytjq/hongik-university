<td width="100%" bordercolor="#000000"><font face="Arial" size="2">Use the following code to copy an ADO field containing an image to another ADO field.</font></td>  </tr>

<table border="0" width="100%" bgcolor="#FFCC00">
  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip Code</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="arial" size="-1" ><p></font><pre>
<font color="#00007F">Private</font> <font color="#00007F">Function</font> GetRandomFileName(sDirTarget <font color="#00007F">As</font> <font color="#00007F">String</font>, _
                                   sPrefix <font color="#00007F">As</font> <font color="#00007F">String</font>, _ 
                                   sExtention <font color="#00007F">As</font> <font color="#00007F">String</font>) <font color="#00007F">As</font> <font color="#00007F">String</font>

<font color="#007F00">'Generates a unique temp file name for the specified directory</font>
<font color="#007F00">'with the specified extention</font>
<font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">Resume</font> <font color="#00007F">Next</font>
<font color="#00007F">Dim</font> fs <font color="#00007F">As</font> FileSystemObject   <font color="#007F00">' **** requires filesystem object</font>
<font color="#00007F">Dim</font> sFName <font color="#00007F">As</font> <font color="#00007F">String</font>
<font color="#00007F">Dim</font> iRnd <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> iUpperBound <font color="#00007F">As</font> <font color="#00007F">Long</font>, iLowerbound <font color="#00007F">As</font> <font color="#00007F">Long</font>

TRYAGAIN:

<font color="#00007F">Randomize</font>
iUpperBound = 99
iLowerbound = 0
iRnd = <font color="#00007F">Int</font>((iUpperBound - iLowerbound + 1) * <font color="#00007F">Rnd</font> + iLowerbound)

sFName = <font color="#00007F">CStr</font>(iRnd) + <font color="#00007F">CStr</font>(<font color="#00007F">DatePart</font>("d", <font color="#00007F">Now</font>())) _
                    + <font color="#00007F">CStr</font>(<font color="#00007F">DatePart</font>("h", <font color="#00007F">Now</font>())) _
                    + <font color="#00007F">CStr</font>(<font color="#00007F">DatePart</font>("n", <font color="#00007F">Now</font>()))

sFName = sPrefix + sFName + "." + sExtention

<font color="#00007F">Set</font> fs = <font color="#00007F">New</font> FileSystemObject

<font color="#00007F">If</font> <font color="#00007F">Not</font> fs.FileExists(sDirTarget + "\" + sFName) <font color="#00007F">Then</font>
    GetRandomFileName = sDirTarget + sFName
    <font color="#00007F">Exit</font> <font color="#00007F">Function</font>
<font color="#00007F">Else</font>
    <font color="#00007F">GoTo</font> TRYAGAIN
<font color="#00007F">End</font> <font color="#00007F">If</font>

<font color="#00007F">End</font> <font color="#00007F">Function</font>

<font color="#00007F">Public</font> <font color="#00007F">Function</font> CopyImageField(fld <font color="#00007F">As</font> ADODB.Field, _
                               fldTO <font color="#00007F">As</font> ADODB.Field)

<font color="#007F00">'This function takes the source field image and copies it</font>
<font color="#007F00">'into the destination field.</font>
<font color="#007F00">'The function first saves the image in the source field to a</font>
<font color="#007F00">'temp file on disc. <font color="#00007F">Then</font> reads this temp file into </font>
<font color="#007F00">'the destination field.</font>
<font color="#007F00">'The temp file is then deleted</font>
<font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">Resume</font> <font color="#00007F">Next</font>

<font color="#00007F">Dim</font> iFieldSize  <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> varChunk    <font color="#00007F">As</font> <font color="#00007F">Variant</font>
<font color="#00007F">Dim</font> baData()    <font color="#00007F">As</font> <font color="#00007F">Byte</font>
<font color="#00007F">Dim</font> iOffset     <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> sFName      <font color="#00007F">As</font> <font color="#00007F">String</font>
<font color="#00007F">Dim</font> iFileNum    <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> cnt         <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> z()         <font color="#00007F">As</font> <font color="#00007F">Byte</font>

<font color="#00007F">Const</font> CONCHUNKSIZE <font color="#00007F">As</font> <font color="#00007F">Long</font> = 16384

<font color="#00007F">Dim</font> iChunks <font color="#00007F">As</font> <font color="#00007F">Long</font>
<font color="#00007F">Dim</font> iFragmentSize <font color="#00007F">As</font> <font color="#00007F">Long</font>
    
    <font color="#007F00">'Get a unique random filename</font>
    sFName = GetRandomFileName(App.Path, "pic", "tmp")
    
    <font color="#007F00">'Open a file to output the data to</font>
    iFileNum = FreeFile
    <font color="#00007F">Open</font> sFName <font color="#00007F">For</font> Binary Access Write <font color="#00007F">Lock</font> Write <font color="#00007F">As</font> iFileNum
    
    <font color="#007F00">'Copy the logo to a variable in chunks.</font>
    iFieldSize = fld.ActualSize
    iChunks = iFieldSize / CONCHUNKSIZE
    iFragmentSize = iFieldSize <font color="#00007F">Mod</font> CONCHUNKSIZE
    <font color="#00007F">If</font> iFragmentSize > 0 <font color="#00007F">Then</font> 
<font color="#007F00">' if there is a frag then write it first, else just use chunk</font>
        <font color="#00007F">ReDim</font> baData(iFragmentSize)
        baData() = fld.GetChunk(iFragmentSize)
        <font color="#00007F">Put</font> iFileNum, , baData()
    <font color="#00007F">End</font> <font color="#00007F">If</font>
    
    <font color="#007F00">'Fragment added; <font color="#00007F">Now</font> write rest of chunks</font>
    <font color="#00007F">ReDim</font> baData(CONCHUNKSIZE)
    <font color="#00007F">For</font> cnt = 1 <font color="#00007F">To</font> iChunks
        baData() = fld.GetChunk(CONCHUNKSIZE)
        <font color="#00007F">Put</font> iFileNum, , baData
    <font color="#00007F">Next</font> cnt
    
    <font color="#007F00">'Close file</font>
    <font color="#00007F">Close</font> iFileNum
    
    <font color="#007F00">'Now we have the file on disk Load the pic from</font>
    <font color="#007F00">'the temp file into the other field</font>
    
    <font color="#00007F">Open</font> sFName <font color="#00007F">For</font> Binary Access Read <font color="#00007F">As</font> #1
    <font color="#00007F">ReDim</font> z(<font color="#00007F">FileLen</font>(sFName))
    <font color="#00007F">Get</font> #1, , z()
    fldTO.AppendChunk z
    <font color="#00007F">Close</font> #1
    
    <font color="#007F00">'Delete the file</font>
    <font color="#00007F">Kill</font> (sFName)
    
<font color="#00007F">End</font> <font color="#00007F">Function</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica"></p></font></td></tr></table>
