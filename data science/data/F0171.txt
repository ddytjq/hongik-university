sub window_onload
i = 0
for each a in document.links
	if document.links(i).name <> "" then
	document.links(i).style.display = "none"
	
	end if

	i = i +1
next

end sub

sub vbrec_onmousemove
for each a in document.links
	if left(document.links(i).name,3) = "vbr" then
	document.links(i).style.display = "block"
	
	end if
if left(document.links(i).name,3) = "vbd" then
	document.links(i).style.display = "none"
end if
	i = i +1
next
end sub



sub vbdown_onmousemove
for each a in document.links

	if left(document.links(i).name,3) = "vbr" then
	document.links(i).style.display = "none"
end if
	if left(document.links(i).name,3) = "vbd" then
	document.links(i).style.display = "block"
	
	end if

	i = i +1
next
end sub

end sub
