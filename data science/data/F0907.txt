  <tr>
<td width="100%"><font face="Arial" size="2"><b>Tip Code</b></font></td>  </tr>
</table>
<br>
<table border="1" width="100%" bordercolor="#FFFFFF">
  <tr>
<td width="100%" bordercolor="#000000"><font face="arial" size="-1" ><p></font><pre>
<font color="#00007F">Public</font> <font color="#00007F">Function</font> WriteSequentialFile(strLineToWrite <font color="#00007F">As</font> <font color="#00007F">String</font>, _ 
                                    strPath <font color="#00007F">As</font> <font color="#00007F">String</font>, _ 
                                    strFileName <font color="#00007F">As</font> <font color="#00007F">String</font>) <font color="#00007F">As</font> <font color="#00007F">Boolean</font>

    <font color="#007F00">'Writes any string out to a sequential file at a specified dir...</font>
    <font color="#007F00">'Will create file and one directory branch if neccessary...</font>
    <font color="#007F00">'you may not specified a non existant</font>
    <font color="#007F00">'directory more than one level deep.  File must have read/write access. </font>
    <font color="#007F00">'Closing \ on directory</font>
    <font color="#007F00">'should not be included in the function call var strPath</font>
    
    <font color="#00007F">Dim</font> FileNum <font color="#00007F">As</font> <font color="#00007F">Integer</font> <font color="#007F00">'Windows File Number Reference</font>
    <font color="#00007F">Dim</font> FileExists <font color="#00007F">As</font> <font color="#00007F">String</font> <font color="#007F00">'String test of File Existance</font>
    <font color="#00007F">Dim</font> DirExists <font color="#00007F">As</font> <font color="#00007F">String</font> <font color="#007F00">'Same for the Directory</font>
    
    <font color="#00007F">On</font> <font color="#00007F">Error</font> <font color="#00007F">GoTo</font> FileAccessError <font color="#007F00">'If a path error occurs ignore the write</font>
    
    FileNum = FreeFile <font color="#007F00">' <font color="#00007F">Get</font> a free file number</font>
    
    DirExists = <font color="#00007F">Dir</font>(strPath, vbDirectory) <font color="#007F00">'Check directory existance</font>
    
    <font color="#00007F">If</font> DirExists = "" <font color="#00007F">Then</font>
        <font color="#00007F">MkDir</font> (strPath) <font color="#007F00">'If its not there then make this path </font>
<font color="#007F00">'(One new dir deep only)</font>
    <font color="#00007F">End</font> <font color="#00007F">If</font>
    
    FileExists = <font color="#00007F">Dir</font>(strPath & "\" & strFileName, vbNormal) 
<font color="#007F00">'Check for file existance</font>
    <font color="#00007F">If</font> FileExists = "" <font color="#00007F">Then</font>
         <font color="#00007F">Open</font> strPath & "\" & strFileName <font color="#00007F">For</font> Output <font color="#00007F">As</font> FileNum 
<font color="#007F00">'If it's not there create it and write the line</font>
            <font color="#00007F">Print</font> #FileNum, <font color="#00007F">Now</font> & <font color="#00007F">Space</font>(2) & strLineToWrite
        <font color="#00007F">Close</font> FileNum
    <font color="#00007F">Else</font>
        <font color="#00007F">Open</font> strPath & "\" & strFileName <font color="#00007F">For</font> Append <font color="#00007F">As</font> FileNum 
<font color="#007F00">'If it is there then append data to end of file</font>
            <font color="#00007F">Print</font> #FileNum, <font color="#00007F">Now</font> & <font color="#00007F">Space</font>(2) & strLineToWrite
        <font color="#00007F">Close</font> FileNum
    <font color="#00007F">End</font> <font color="#00007F">If</font>
    
<font color="#00007F">Exit</font> <font color="#00007F">Function</font>   

FileAccessError:
<font color="#007F00">'Handle File Access <font color="#00007F">Error</font></font>
        
<font color="#00007F">End</font> <font color="#00007F">Function</font>
</pre><FONT SIZE="-1" FACE="verdana, arial, helvetica"></p></font></td></tr></table>
