							msg = msg + ", Phone Number";
						}
					else
						{
							msg = "Phone Number";
						}
				}
				if (document.forms['form2'].txtEmail.value == "")
				{
					if (msg != null)
						{
							msg = msg + ", Email Address";
						}
					else
						{
							msg = "Email Address"
						}
				}
				if (document.forms['form2'].txtCompany.value == "")
				{
					if (msg != null)
						{
							msg = msg + ", Company";
						}
					else
						{
							msg = "Company";
						}
				}
				
				if (document.forms['form2'].txtAddress.value == "")

							msg = "Address";
